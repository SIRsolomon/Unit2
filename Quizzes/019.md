```.py
import random
random.seed(123456789)

def produce(n,m,s):
    if s == 0:
        out = 'invalid'
    else:
        out = "| x | y(x) |\n"
        for i in range (n):
            z = ((m/s)**2)/2

            x = int(100*(random.random()))
            y = x**z
            out += f'| {x} | {y:.2f} |\n'

    return out

sample = produce(5,3,2)
print(sample)
```
