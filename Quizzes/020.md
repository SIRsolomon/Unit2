```.py
import random
from matplotlib import pyplot as plt

def produce(n,m,s):
    X = []
    Y = []
    if s == 0:
        out = 'invalid'
    else:
        out = "| x | y(x) |\n"
        for i in range (n):
            z = ((m/s)**2)/2
            x = int(100*(random.random()))
            y = x**z
            out += f'| {x} | {y:.2f} |\n'
            Y += [y]
            X += [x]
    return Y, X

y, x = produce(5,3,2)
print(f'{x},{y}')
plt.plot(x,y,'or')
plt.xlabel('X')
plt.ylabel('Y')
plt.title('y(x)')
plt.show()
```
