def get_Truth(n:int):
    digits = []
    for i in range (0, n):
        digits = digits + [0]
    o = ''
    for i in range (0, n):
        o = o + f'| bit{n-1-i} |'

    for i in range (0, 2**n):
        for b in range (n):
            if i%(2**n):
                digits[n-1-b] = not digits[n-1-b]
        for i in range(0, n):
            o = o + str(digits[i])
    return o

def get_TTruth(n: int):
    digits = [0] * n
    head = ""
    for i in range(n):
        head += f"| bit{n - 1 - i} "
    head += "|"

    rows = [head]

    for i in range(2 ** n):
        row = "| "
        for digit in digits:
            row += str(digit) + " | "
        rows.append(row)

        for b in range(n):
            if (i + 1) % (2 ** b) == 0:
                digits[n - 1 - b] = 1 - digits[n - 1 - b]

    return "\n".join(rows)

out = get_TTruth(3)
